Complete CI/CD Lab Manual — Maven + Jenkins Setup
This manual explains step-by-step how to: - Create a Maven Java project - Configure Maven and JDK - Install Jenkins
- Push project to GitHub - Create and run a CI/CD pipeline successfully
Step 1 — Check Java Installation
java -version
Step 2 — Install Maven
sudo apt update
sudo apt install maven -y
Step 3 — Go to Home Directory
cd ~
Step 4 — Delete Old Project
rm -rf demo-app
Step 5 — Create Maven Project
mvn archetype:generate -DgroupId=com.example.app \
-DartifactId=demo-app \
-DarchetypeArtifactId=maven-archetype-quickstart \
-DinteractiveMode=false
Step 6 — Enter Project Folder
cd demo-app
Step 7 — Verify Files
ls


Step 8 — Edit pom.xml
<project xmlns="http://maven.apache.org/POM/4.0.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

Step 9 — Create App.java
package com.example.app;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.apache.commons.lang3.StringUtils;
public class App {
 private static final Logger logger = LoggerFactory.getLogger(App.class);
 public static void main(String[] args) {
 String name = "CI/CD Pipeline";
 if (StringUtils.isNotBlank(name)) {
 String message = greet(name);
 logger.info(message);
 System.out.println(message);
 } else {
 logger.error("Name is empty!");
 }
 }
 public static String greet(String name) {
 return "Hello, " + name + "! Welcome to Maven CI/CD Demo.";
 }
}




Step 10 — Create AppTest.java
package com.example.app;
import org.junit.Test;
import static org.junit.Assert.assertEquals;
public class AppTest {
 @Test
 public void testGreet() {
 String result = App.greet("Student");
 assertEquals(
 "Hello, Student! Welcome to Maven CI/CD Demo.",
 result
 );
 }
}

Step 11 — Build and Run Project
mvn clean install
mvn exec:java -Dexec.mainClass="com.example.app.App"
Step 12 — Install Jenkins
sudo apt install fontconfig openjdk-21-jre -y
wget https://pkg.jenkins.io/debian-stable/binary/jenkins_2.504.1_all.deb
sudo dpkg -i jenkins_2.504.1_all.deb
sudo systemctl start jenkins
sudo systemctl enable jenkins
Step 13 — Create Jenkinsfile
pipeline {
 agent any
 tools {
 maven 'Maven'
 jdk 'JDK21'
 }
 stages {
 stage('Checkout') {
 steps {
 git branch: 'main',
 url: 'https://github.com/20divine05/lab5-pp.git'
 }
 }
 stage('Build') {
 steps {
 sh 'mvn clean compile'
 }
 }
 stage('Test') {
 steps {
 sh 'mvn test'
 }
 }
 stage('Package') {
 steps {
 sh 'mvn package'
 }
 }
 stage('Run Application') {
 steps {
 sh 'mvn exec:java -Dexec.mainClass="com.example.app.App"'
 }
 }
 }
}


Step 14 — Configure Jenkins Pipeline
1. Open Jenkins → http://localhost:8080
2. Manage Jenkins → Tools
3. Add JDK:
 Name: JDK21
 Install Automatically: Enabled
4. Add Maven:
 Name: Maven
 Install Automatically: Enabled
5. Create New Item → Pipeline
6. Configure:
 - Pipeline script from SCM
 - Git
 - Repository URL
 - Branch: */main
 - Script Path: Jenkinsfile
7. Click Build Now
Final Expected Output
Expected Final Output:
Hello, CI/CD Pipeline! Welcome to Maven CI/CD Demo.
BUILD SUCCESS
Finished: SUCCESS
